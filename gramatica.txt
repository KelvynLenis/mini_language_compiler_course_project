Correções 
1) declaracoes in-line
2) precedencia de operadores
3) operadores booleanos/relacionais

=========================================================================

programa : 
	'begin' declarationList
	'do' commandList 'end'

declarationList : 
	declaration declarationList | declaration;

declaration : 
	varType VARIAVEL;

varType : 
	'int' | 'float';

commandList : 
	command commandList | command;

command : 
	assignCommand | 
	outputCommand | 
	conditionCommand | 
	loopCommand

assignCommand : 
	VARIAVEL '=' arithmeticExpression;
	
outputCommand : 
	'print'  VARIAVEL; | (cadeia);

cadeia:
	QUALQUER_IDENTIFICADOR

conditionCommand:
	'if' relationalExpression 'then' command conditionCommand2;

conditionCommand2:
	'else' command | VAZIO

loopCommand : 'while' relationalExpression command;

arithmeticExpression:
	arithmeticTerm arithmeticExpression2

arithmeticExpression2:
	arithmeticExpression3 arithmeticExpression2 |
	VAZIO

arithmeticExpression3:
	'+' arithmeticTerm |
	'-' arithmeticTerm 

arithmeticTerm : 
	arithmeticTerm '*' arithmeticFactor |
	arithmeticTerm '/' arithmeticFactor |
	arithmeticFactor;

arithmeticTerm : 
	arithmeticFactor arithmeticTerm2;

arithmeticTerm2: 
	'*' arithmeticFactor arithmeticTerm2 |
	'/' arithmeticFactor arithmeticTerm2 |
	VAZIO;

arithmeticFactor :
	NUMINT | NUMREAL | VARIAVEL |
	'(' arithmeticExpression ')'

relationalExpression:
	relationalTerm relationalExpression2;

relationalExpression2:
	booleanOperator relationalTerm relationalExpression2 | VAZIO

relationalTerm : 
	arithmeticExpression relationalOperator arithmeticExpression |
	'(' relationalExpression ')'

booleanOperator : 
	'and' | 'or'

relationalOperator:
	'>' | '<' | '>=' | '<=' | '==' | '!='

=========================================================================
